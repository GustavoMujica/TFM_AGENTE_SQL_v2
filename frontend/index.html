SQL generado ▲ ocultar 
SELECT u.id, u.name, SUM(o.total) AS total_gastado
FROM users u
INNER JOIN orders o ON u.id = o.user_id
GROUP BY u.id, u.name
HAVING SUM(o.total) > (
    SELECT AVG(user_total)
    FROM (SELECT SUM(total) AS user_total
          FROM orders
          GROUP BY user_id) AS subq
)
